1function [out R] = cindir(q, A0b, Ane)
 2
 3%Calcola la cinematica diretta dello SmartSix
 4%Input:
 5%      qDH; variabili di giunti in convenzione DH e in rad
 6%Output:
 7%       x:     posizione x organo terminale
 8%       y:     posizione y organo terminale
 9%       z:     posizione z organo terminale
10%       phi:   orientamento phi organo terminale in rad
11%       theta: orientamento theta organo terminale in rad
12%       psi:   orientamento psi organo terminale in rad
13%varargin: previous vector for orientation euler angles correction
14if(size(q, 2) ~= 8)
15    qDHn = q';
16else
17    qDHn = q;
18end
19
20out = zeros(size(qDHn,1),6);
21
22[a, d, alpha] = dh_table;
23
24for i=1:size(qDHn,1)
25    qDH = qDHn(i, :);
26    ctheta = cos(qDH);
27    stheta = sin(qDH);
28    calpha = cos(alpha);
29    salpha = sin(alpha);
30    
31    T = eye(4);
32    if(length(qDH)>6)
33        T(1,4) = T(1,4) +  qDH(7);
34        T(2,4) = T(2,4) +  qDH(8);
35        T(3,4) = 0; %0.2985;
36    end
37    T = A0b*T;
38    for j=1:6
39        T =T*[ctheta(j) -stheta(j)*calpha(j) stheta(j)*salpha(j) a(j)*ctheta(j);
40            stheta(j)  ctheta(j)*calpha(j) -ctheta(j)*salpha(j) a(j)*stheta(j);
41            0           salpha(j)             calpha(j)            d(j);
42            0                0                  0                    1];
43    end
44    T = T*Ane;
45    x = T(1, 4);
46    y = T(2, 4);
47    z = T(3, 4);
48    [phi, theta, psi] = R2ZYZ(T(1:3, 1:3));
49    R = T(1:3, 1:3);
50    out(i, :) = [x y z phi theta psi]';
51    
52    
53    %
54    if i>1
55        for k=4:6
56            if abs(out(i, k)-out(i-1, k))>0.5
57                out(i,k) = out(i, k)+sign(out(i-1, k)-out(i, k))*round(abs(out(i, k)-out(i-1, k))/(pi))*(pi);
58            end
59        end
60    end
61end
