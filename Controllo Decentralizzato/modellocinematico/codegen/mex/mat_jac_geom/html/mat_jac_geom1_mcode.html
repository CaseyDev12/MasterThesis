1function [J] = mat_jac_geom(q, A0b, Ane)
 2%Calcola lo Jacobiano geometrico dello Smart Six non in simbolico
 3%Input:
 4%      qDH; variabili di giunti in convenzione DH e in rad
 5%Output:
 6%       J: jacobiano geometrico
 7
 8%Marino: 09/03/2010
 9
10
11if(size(q, 2) ~= 8)
12    qDH = q';
13else
14    qDH = q;
15end
16    
17
18[a, d, alpha] = dh_table;
19
20
21%seni e coseni degli angoli
22ctheta = cos(qDH);
23stheta = sin(qDH);
24calpha = cos(alpha);
25salpha = sin(alpha);
26
27
28     T = eye(4);
29     if(length(qDH)>6)
30     T(1,4) = T(1,4) +  0*qDH(7);
31     T(2,4) = T(2,4) +  0*qDH(8);
32     T(3,4) = 0; %0.2985;
33     end
34     %T = A0b*T; 
35% T = eye(4); %qDH(7:8) = 0;
36temp = qDH;
37temp(7:8) = 0;
38p = zeros(6,1);
39p(:, 1) = cindir(temp,  eye(4), Ane); %cinematica diretta
40
41% zi = T(1:3, 3); pi = T(1:3, 4);
42% J = [cross(zi, (p-pi); zi];
43J = zeros(6,length(qDH));
44for j=1:6
45    zi = T(1:3, 3); pi = T(1:3, 4);
46    a_ = zi;
47    b_ = (p(1:3)-pi);
48    prodvett = [a_(2,:).*b_(3,:)-a_(3,:).*b_(2,:)
49     a_(3,:).*b_(1,:)-a_(1,:).*b_(3,:)
50     a_(1,:).*b_(2,:)-a_(2,:).*b_(1,:)];
51 %cross(zi, (p(1:3)-pi))
52    J(:,j) = [prodvett; zi];
53    
54    T =T*[ctheta(j) -stheta(j)*calpha(j) stheta(j)*salpha(j) a(j)*ctheta(j);
55          stheta(j)  ctheta(j)*calpha(j) -ctheta(j)*salpha(j) a(j)*stheta(j);
56             0           salpha(j)             calpha(j)            d(j);
57             0                0                  0                    1];
58
59end
60
61% temp(:,1) = [;
62%J(1:2, 7) = A0b(1:2,1); J(1:2, 8) = A0b(1:2,2);
63 if(length(qDH)>6)
64 J(1,7) = 1;
65 J(2,8) = 1;
66 end
67 R0b = A0b(1:3, 1:3);
68 J = blkdiag(R0b, R0b)*J;
