1function qdh = comau2dh(qcomau)
 2%
 3% SIX - Gennaio 2005
 4%
 5% qdh = comau2dh(qcomau)
 6%
 7% Converte le posizioni dei giunti dalla convenzione COMAU alla convenzione DH
 8%
 9% Ingressi:
10%   qcomau: vettore 6x1 delle posizioni in convenzione COMAU in gradi
11% Uscite:
12%   qdh   : vettore 6x1 delle posizioni in convenzione DH in gradi
13% Note:
14%   le unita' di misura devono essere necessariamente in gradi
15%   la chiamata seguente: comau2dh(zeros(6,1)) fornisce il vettore q0 in gradi in convenzione DH
16qdh = zeros(6,1);
17qdh(1) = -qcomau(1) +   0;
18qdh(2) =  qcomau(2) -  90;
19qdh(3) = -qcomau(3) -  90;
20qdh(4) = -qcomau(4) +   0;
21qdh(5) =  qcomau(5) +   0;
22qdh(6) = -qcomau(6) + 180;
23if(length(qcomau)>6)
24    qdh(7) = qcomau(7);
25end
