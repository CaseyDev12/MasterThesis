1function dqdh = comau2dh_vel(dqcomau)
 2%
 3% SIX - Gennaio 2005
 4%
 5% dqdh = comau2dh_vel(dqcomau)
 6%
 7% Converte le velocità/accelerazioni dei giunti dalla convenzione COMAU alla convenzione DH
 8%
 9% Ingressi:
10%   dqcomau: vettore 6x1 delle velocità (o accelerazioni) in convenzione COMAU
11% Uscite:
12%   dqdh   : vettore 6x1 delle velocità (o accelerazioni) in convenzione DH
13% Note:
14%   le unità di misura sono ininfluenti
15
16if(length(dqcomau)>6)
17dqdh = zeros(7,1);
18else
19dqdh = zeros(6,1);  
20end
21
22dqdh(1) = -dqcomau(1);
23dqdh(2) =  dqcomau(2);
24dqdh(3) = -dqcomau(3);
25dqdh(4) = -dqcomau(4);
26dqdh(5) =  dqcomau(5);
27dqdh(6) = -dqcomau(6);
28if(length(dqcomau)>6)
29    dqdh(7) = dqcomau(7);
30end
