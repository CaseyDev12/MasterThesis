1function [dqsigma, x, dq0] = InverseKinematicsFirstOrder(dsig_des, sig_des, eulAglePrec)
 2
 3global Jsig N robot ksigma invJsig
 4
 5x = zeros(6*N,1);
 6for i=1:N
 7    % cindir_mex a partire dalle q attuali del robot, quelle vere, registrate dopo
 8    % il controllo
 9    x(1+(i-1)*6:6+(i-1)*6,1) = cindir(robot(i).qActual, robot(i).A, robot(i).Ane);
10    x(4+(i-1)*6:6+(i-1)*6,1) = AngleCorrectionTask(eulAglePrec(:,i), x(4+(i-1)*6:6+(i-1)*6,1));
11end
12
13sigma = Jsig*x;
14gamma = dsig_des + ksigma*(sig_des - sigma);
15ustim = invJsig*gamma;
16dxstim = ustim;
17
18dqsigma = zeros(N,8);
19dq0 = zeros(N,8);
20for i=1:N
21    % Calcolo dqsigmai
22    Ja = robot(i).Ja;
23    pinvJa = robot(i).pinvJa;
24    dq0(i,:) = NulloFirstOrder(robot(i).qActual, pinvJa, Ja);
25    dqsigma(i,:) = pinvJa*(robot(i).gamma*ustim) + dq0(i,:)';
26end
27
28end
